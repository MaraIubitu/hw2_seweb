# SPARQL Queries for Book Recommendation System Ontology
# These queries are executed GraphDB, screenshots can be found in the repo

# Query 1: List all books and their titles
PREFIX br: <http://example.org/book-recommendation#>

SELECT ?book ?title
WHERE {
  ?book a br:Book .
  ?book br:title ?title .
}

# Query 2: List all books with their authors
PREFIX br: <http://example.org/book-recommendation#>

SELECT ?book ?title ?author
WHERE {
  ?book a br:Book .
  ?book br:title ?title .
  ?book br:hasAuthor ?author .
}

# Query 3: List all books with their genres
PREFIX br: <http://example.org/book-recommendation#>

SELECT ?book ?title ?genre
WHERE {
  ?book a br:Book .
  ?book br:title ?title .
  ?book br:hasGenre ?genre .
}

# Query 4: List all books with their reviews
PREFIX br: <http://example.org/book-recommendation#>

SELECT ?book ?title ?review
WHERE {
  ?book a br:Book .
  ?book br:title ?title .
  ?book br:hasReview ?review .
}

# Query 5: List all books published after 1900
PREFIX br: <http://example.org/book-recommendation#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

SELECT ?book ?title ?year
WHERE {
  ?book a br:Book .
  ?book br:title ?title .
  ?book br:publicationYear ?year .
  FILTER(xsd:decimal(?year) > 1900)
}
