# SPARQL Queries for Book Recommendation System Ontology
# These queries are executed in Protégé or GraphDB

# Query 1: Find all books by a specific author
# Returns all books written by Frank Herbert
SELECT ?bookTitle ?bookAuthor ?bookTheme ?bookLevel
WHERE {
  ?book rdf:type <http://example.org/book/Book> .
  ?book <http://www.w3.org/2000/01/rdf-schema#label> ?bookTitle .
  ?book <http://example.org/book/author> ?bookAuthor .
  ?book <http://example.org/book/hasTheme> ?themeUri .
  ?themeUri <http://www.w3.org/2000/01/rdf-schema#label> ?bookTheme .
  ?book <http://example.org/book/hasReadingLevel> ?levelUri .
  ?levelUri <http://www.w3.org/2000/01/rdf-schema#label> ?bookLevel .
  FILTER (?bookAuthor = "Frank Herbert")
}
ORDER BY ?bookTitle

# Query 2: Find all books with a specific theme
# Returns all Science Fiction books
SELECT ?bookTitle ?bookAuthor ?bookLevel
WHERE {
  ?book rdf:type <http://example.org/book/Book> .
  ?book <http://www.w3.org/2000/01/rdf-schema#label> ?bookTitle .
  ?book <http://example.org/book/author> ?bookAuthor .
  ?book <http://example.org/book/hasTheme> ?themeUri .
  ?themeUri <http://www.w3.org/2000/01/rdf-schema#label> ?themeName .
  ?book <http://example.org/book/hasReadingLevel> ?levelUri .
  ?levelUri <http://www.w3.org/2000/01/rdf-schema#label> ?bookLevel .
  FILTER (?themeName = "Science Fiction")
}
ORDER BY ?bookLevel

# Query 3: Find recommended books for a specific user
# Returns books recommended for Bob (Mystery theme, Beginner level)
SELECT ?bookTitle ?bookAuthor ?bookLevel
WHERE {
  ?user rdf:type <http://example.org/book/User> .
  ?user <http://www.w3.org/2000/01/rdf-schema#label> "Bob" .
  ?user <http://example.org/book/prefers> ?userThemeUri .
  ?user <http://example.org/book/hasReadingLevel> ?userLevelUri .
  ?userLevelUri <http://www.w3.org/2000/01/rdf-schema#label> ?userLevel .

  ?book rdf:type <http://example.org/book/Book> .
  ?book <http://www.w3.org/2000/01/rdf-schema#label> ?bookTitle .
  ?book <http://example.org/book/author> ?bookAuthor .
  ?book <http://example.org/book/hasTheme> ?userThemeUri .
  ?book <http://example.org/book/hasReadingLevel> ?bookLevelUri .
  ?bookLevelUri <http://www.w3.org/2000/01/rdf-schema#label> ?bookLevel .

  FILTER (?bookLevel = ?userLevel)
}

# Query 4: Find all books with multiple themes
# Returns books that have more than one theme
SELECT ?bookTitle (COUNT(?theme) as ?themeCount)
WHERE {
  ?book rdf:type <http://example.org/book/Book> .
  ?book <http://www.w3.org/2000/01/rdf-schema#label> ?bookTitle .
  ?book <http://example.org/book/hasTheme> ?theme .
}
GROUP BY ?bookTitle
HAVING (COUNT(?theme) > 1)
ORDER BY DESC(?themeCount)

# Query 5: Get all users and their preferred themes
# Returns all users and the themes they prefer
SELECT ?userName ?themeName ?readingLevel
WHERE {
  ?user rdf:type <http://example.org/book/User> .
  ?user <http://www.w3.org/2000/01/rdf-schema#label> ?userName .
  ?user <http://example.org/book/prefers> ?themeUri .
  ?themeUri <http://www.w3.org/2000/01/rdf-schema#label> ?themeName .
  ?user <http://example.org/book/hasReadingLevel> ?levelUri .
  ?levelUri <http://www.w3.org/2000/01/rdf-schema#label> ?readingLevel .
}
ORDER BY ?userName
